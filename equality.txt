bool equalityTriangle(Triangle t1, Triangle t2)
{
	if (t1.a == t2.a || t1.b == t2.b) {
		if (t1.c == t2.c)
			return true;
	}
	if (t1.a == t2.b || t1.b == t2.c) {
		if (t1.c == t2.a)
			return true;
	}
	if (t1.a == t2.c || t1.b == t2.a) {
		if (t1.c == t2.b)
			return true;
	}
	else
		return false;
}
